  public: 存放静态文件
  routes: 存放路由文件
    index.js 
    posts.js
    comments.js
    signin.js
    signup.js
    signout.js
  views: 存放模板文件
    components
    404.ejs
    footer.ejs
    header.ejs
    post.ejs
    posts.ejs
    signin.ejs
    signup.ejs
  index.js: 程序主文件
  package.json: 项目依赖、描述、入口等信息
